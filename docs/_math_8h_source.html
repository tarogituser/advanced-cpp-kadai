181 static Quaternion Euler(float pitchDegree, float yawDegree, float rollDegree) noexcept
182 {
183 // pitch = X, yaw = Y, roll = Z
184 // “かけ順” は Unity ライクに Z->X->Y（ZXY）で合成する
185 using namespace DirectX;
186 const float pitch = pitchDegree * Deg2Rad; // X
187 const float yaw = yawDegree * Deg2Rad; // Y
188 const float roll = rollDegree * Deg2Rad; // Z
189 const XMVECTOR qz = XMQuaternionRotationAxis(g_XMIdentityR2, roll);
190 const XMVECTOR qx = XMQuaternionRotationAxis(g_XMIdentityR0, pitch);
191 const XMVECTOR qy = XMQuaternionRotationAxis(g_XMIdentityR1, yaw);
192
193 const XMVECTOR q = XMQuaternionMultiply(XMQuaternionMultiply(qz, qx), qy);
194 return Quaternion(q);
195 }

197 static Quaternion FromToRotation(const Vector3& fromDir, const Vector3& toDir) noexcept
198 {
199 using namespace DirectX;
200 const XMVECTOR F = XMVector3Normalize(XMLoadFloat3(&fromDir));
201 const XMVECTOR T = XMVector3Normalize(XMLoadFloat3(&toDir));
202 const float dot = XMVectorGetX(XMVector3Dot(F, T));
203 if (dot >= 1.f)
204 {
205 return identity;
206 }
207 else if (dot <= -1.f)
208 {
209 XMVECTOR axis = XMVector3Cross(F, XMLoadFloat3(&Vector3::right));
210 if (XMVector3NearEqual(XMVector3LengthSq(axis), g_XMZero, g_XMEpsilon))
211 {
212 axis = XMVector3Cross(F, XMLoadFloat3(&Vector3::up));
213 }
214
215 const XMVECTOR Q = XMQuaternionRotationAxis(axis, XM_PI);
216 return Quaternion(Q);
217 }
218 else
219 {
220 const XMVECTOR C = XMVector3Cross(F, T);
221 XMFLOAT4 result;
222 XMStoreFloat4(&result, C);
223 const float s = sqrtf((1.f + dot) * 2.f);
224 result.x /= s;
225 result.y /= s;
226 result.z /= s;
227 result.w = s * 0.5f;
228 return Quaternion(result);
229 }
230 }

232 static Quaternion LookRotation(const Vector3& forward, const Vector3& up) noexcept
233 {
234 using namespace DirectX;
235 Quaternion q1 = FromToRotation(Vector3::forward, forward);
236 Vector3 c = Cross(forward, up.normalized());
237 const XMVECTOR C = XMLoadFloat3(&c);
238 if (XMVector3NearEqual(XMVector3LengthSq(C), g_XMZero, g_XMEpsilon))
239 {
240 return q1;
241 }
242 const XMVECTOR U = XMVector3Rotate(XMLoadFloat3(&Vector3::up), q1.XMLoad());
243 Quaternion q2 = FromToRotation(Vector3(U), up);
244 return Quaternion(XMQuaternionMultiply(q2.XMLoad(), q1.XMLoad()));
245 }
