192 static Quaternion Euler(float pitchDegree, float yawDegree, float rollDegree) noexcept
193 {
194 // pitch = X, yaw = Y, roll = Z
195 // “かけ順” は Unity ライクに Z->X->Y（ZXY）で合成する
196 using namespace DirectX;
197 const float pitch = pitchDegree * Deg2Rad; // X
198 const float yaw = yawDegree * Deg2Rad; // Y
199 const float roll = rollDegree * Deg2Rad; // Z
200 const XMVECTOR qz = XMQuaternionRotationAxis(g_XMIdentityR2, roll);
201 const XMVECTOR qx = XMQuaternionRotationAxis(g_XMIdentityR0, pitch);
202 const XMVECTOR qy = XMQuaternionRotationAxis(g_XMIdentityR1, yaw);
203
204 const XMVECTOR q = XMQuaternionMultiply(XMQuaternionMultiply(qz, qx), qy);
205 return Quaternion(q);
206 }

208 static Quaternion FromToRotation(const Vector3& fromDir, const Vector3& toDir) noexcept
209 {
210 using namespace DirectX;
211 const XMVECTOR F = XMVector3Normalize(XMLoadFloat3(&fromDir));
212 const XMVECTOR T = XMVector3Normalize(XMLoadFloat3(&toDir));
213 const float dot = XMVectorGetX(XMVector3Dot(F, T));
214 if (dot >= 1.f)
215 {
216 return identity;
217 }
218 else if (dot <= -1.f)
219 {
220 XMVECTOR axis = XMVector3Cross(F, XMLoadFloat3(&Vector3::right));
221 if (XMVector3NearEqual(XMVector3LengthSq(axis), g_XMZero, g_XMEpsilon))
222 {
223 axis = XMVector3Cross(F, XMLoadFloat3(&Vector3::up));
224 }
225
226 const XMVECTOR Q = XMQuaternionRotationAxis(axis, XM_PI);
227 return Quaternion(Q);
228 }
229 else
230 {
231 const XMVECTOR C = XMVector3Cross(F, T);
232 XMFLOAT4 result;
233 XMStoreFloat4(&result, C);
234 const float s = sqrtf((1.f + dot) * 2.f);
235 result.x /= s;
236 result.y /= s;
237 result.z /= s;
238 result.w = s * 0.5f;
239 return Quaternion(result);
240 }
241 }

243 static Quaternion LookRotation(const Vector3& forward, const Vector3& up) noexcept
244 {
245 using namespace DirectX;
246 Quaternion q1 = FromToRotation(Vector3::forward, forward);
247 Vector3 c = Cross(forward, up.normalized());
248 const XMVECTOR C = XMLoadFloat3(&c);
249 if (XMVector3NearEqual(XMVector3LengthSq(C), g_XMZero, g_XMEpsilon))
250 {
251 return q1;
252 }
253 const XMVECTOR U = XMVector3Rotate(XMLoadFloat3(&Vector3::up), q1.XMLoad());
254 Quaternion q2 = FromToRotation(Vector3(U), up);
255 return Quaternion(XMQuaternionMultiply(q2.XMLoad(), q1.XMLoad()));
256 }
